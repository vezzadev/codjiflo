> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Enable iteration tracking for PRs without the CodjiFlo GitHub Action by using GitHub's native Timeline and Commits APIs.

**Architecture:** Commit-based iteration loader fetches PR commits and timeline events, builds iteration list with force-push detection. Web Worker with Comlink handles async diff computation. Priority scheduler ensures user-selected files load first. IndexedDB persists last-seen state.

```typescript
// src/lib/tracing/span.ts
import type { Attributes, AttributeValue } from './attributes';

export interface SpanContext {
  traceId: string;
  spanId: string;
  parentSpanId?: string;
}

export interface SpanEvent {
  name: string;
  timestamp: number;
  attributes?: Attributes;
}

export interface Span {
  setAttribute(key: string, value: AttributeValue): void;
  addEvent(name: string, attributes?: Attributes): void;
  setStatus(status: 'ok' | 'error', message?: string): void;
  getContext(): SpanContext;
  end(): void;
}

function generateId(): string {
  return Math.random().toString(36).substring(2, 10);
}

export class ConsoleSpan implements Span {
  private readonly context: SpanContext;
  private readonly name: string;
  private readonly startTime: number;
  private readonly attributes: Attributes;
  private readonly events: SpanEvent[] = [];
  private status: 'ok' | 'error' = 'ok';
  private statusMessage?: string;

  constructor(name: string, attributes: Attributes = {}, parentContext?: SpanContext) {
    this.name = name;
    this.startTime = Date.now();
    this.attributes = { ...attributes };
    this.context = {
      traceId: parentContext?.traceId ?? generateId(),
      spanId: generateId(),
      parentSpanId: parentContext?.spanId,
    };
  }

  setAttribute(key: string, value: AttributeValue): void {
    this.attributes[key] = value;
  }

  addEvent(name: string, attributes?: Attributes): void {
    this.events.push({
      name,
      timestamp: Date.now(),
      attributes,
    });
  }

  setStatus(status: 'ok' | 'error', message?: string): void {
    this.status = status;
    this.statusMessage = message;
  }

  getContext(): SpanContext {
    return this.context;
  }

  end(): void {
    const data = {
      timestamp: this.startTime,
      traceId: this.context.traceId,
      spanId: this.context.spanId,
      parentSpanId: this.context.parentSpanId,
      name: this.name,
      duration_ms: Date.now() - this.startTime,
      attributes: this.attributes,
      events: this.events,
      status: this.status,
      statusMessage: this.statusMessage,
    };
    console.debug(JSON.stringify(data));
  }
}
```

```typescript
// src/features/iterations/loaders/timeline-loader.ts
import { tracer, SemanticAttributes as Attr } from '@/lib/tracing';
import type { ForcePushEvent } from '../types';

interface TimelineEvent {
  event: string;
  before?: { sha: string };
  after?: { sha: string };
  created_at?: string;
  actor?: { login: string };
}

function parseNextLink(linkHeader: string | null): string | null {
  if (!linkHeader) return null;

  const matches = linkHeader.match(/<([^>]+)>;\s*rel="next"/);
  return matches?.[1] ?? null;
}

export async function loadTimeline(
  owner: string,
  repo: string,
  prNumber: number,
  token?: string
): Promise<ForcePushEvent[]> {
  const span = tracer.startSpan('timeline.load', {
    [Attr.GITHUB_OWNER]: owner,
    [Attr.GITHUB_REPO]: repo,
    [Attr.GITHUB_PR_NUMBER]: prNumber,
  });

  const events: ForcePushEvent[] = [];
  let url: string | null =
    `https://api.github.com/repos/${owner}/${repo}/issues/${prNumber}/timeline?per_page=100`;

  try {
    while (url) {
      span.addEvent('page.fetch', { [Attr.HTTP_URL]: url });

      const headers: HeadersInit = {
        Accept: 'application/vnd.github+json',
      };
      if (token) {
        headers.Authorization = `Bearer ${token}`;
      }

      const response = await fetch(url, { headers });

      if (!response.ok) {
        throw new Error(`Timeline API error: ${response.status} ${response.statusText}`);
      }

      span.setAttribute(Attr.HTTP_STATUS_CODE, response.status);

      const timeline: TimelineEvent[] = await response.json();

      for (const item of timeline) {
        if (
          item.event === 'head_ref_force_pushed' &&
          item.before?.sha &&
          item.after?.sha &&
          item.created_at &&
          item.actor?.login
        ) {
          events.push({
            beforeSha: item.before.sha,
            afterSha: item.after.sha,
            timestamp: new Date(item.created_at),
            actor: item.actor.login,
          });
        }
      }

      url = parseNextLink(response.headers.get('Link'));
    }

    span.addEvent('force_push_events.extracted', { 'event.count': events.length });
    span.setStatus('ok');
    return events;
  } catch (error) {
    span.setStatus('error', error instanceof Error ? error.message : 'Unknown error');
    throw error;
  } finally {
    span.end();
  }
}
```

```typescript
// src/features/iterations/loaders/commit-loader.ts
import { tracer, SemanticAttributes as Attr } from '@/lib/tracing';

export interface PRCommit {
  sha: string;
  message: string;
  author: string;
  createdAt: Date;
  parentSha: string;
}

interface GitHubCommit {
  sha: string;
  commit: {
    message: string;
    author: { date: string; name?: string };
  };
  author: { login: string } | null;
  parents: { sha: string }[];
}

function parseNextLink(linkHeader: string | null): string | null {
  if (!linkHeader) return null;
  const matches = linkHeader.match(/<([^>]+)>;\s*rel="next"/);
  return matches?.[1] ?? null;
}

export async function loadCommits(
  owner: string,
  repo: string,
  prNumber: number,
  token?: string
): Promise<PRCommit[]> {
  const span = tracer.startSpan('commits.load', {
    [Attr.GITHUB_OWNER]: owner,
    [Attr.GITHUB_REPO]: repo,
    [Attr.GITHUB_PR_NUMBER]: prNumber,
  });

  const commits: PRCommit[] = [];
  let url: string | null =
    `https://api.github.com/repos/${owner}/${repo}/pulls/${prNumber}/commits?per_page=100`;

  try {
    while (url) {
      span.addEvent('page.fetch', { [Attr.HTTP_URL]: url });

      const headers: HeadersInit = {
        Accept: 'application/vnd.github+json',
      };
      if (token) {
        headers.Authorization = `Bearer ${token}`;
      }

      const response = await fetch(url, { headers });

      if (!response.ok) {
        throw new Error(`Commits API error: ${response.status} ${response.statusText}`);
      }

      span.setAttribute(Attr.HTTP_STATUS_CODE, response.status);

          createdAt: new Date(commit.commit.author.date),
          parentSha: commit.parents[0]?.sha ?? '',
        });

      url = parseNextLink(response.headers.get('Link'));
```

```typescript
// src/features/iterations/loaders/iteration-builder.ts
import { tracer, SemanticAttributes as Attr } from '@/lib/tracing';
import type {
  StatelessIteration,
  CollapsedIterationGroup,
  ForcePushEvent,
  StatelessIterationData,
} from '../types';
import type { PRCommit } from './commit-loader';

export function buildStatelessIterations(
  liveCommits: PRCommit[],
  forcePushEvents: ForcePushEvent[],
  prBaseSha: string,
  discardedCommits: PRCommit[] = []
): StatelessIterationData {
  const span = tracer.startSpan('iterations.build', {
    'live_commit.count': liveCommits.length,
    'force_push.count': forcePushEvents.length,
    'discarded_commit.count': discardedCommits.length,
  });

  try {
    const iterations: StatelessIteration[] = [];
    const collapsedGroups: CollapsedIterationGroup[] = [];

    // Sort force-push events by timestamp
    const sortedForcePushes = [...forcePushEvents].sort(
      (a, b) => a.timestamp.getTime() - b.timestamp.getTime()
    );

    // Build a map of discarded commits by SHA
    const discardedBySha = new Map(discardedCommits.map((c) => [c.sha, c]));

    // Track which commits belong to which force-push group
    const commitToGroup = new Map<string, string>();

    // Create collapsed groups for each force-push
    for (const event of sortedForcePushes) {
      const groupId = `group-${event.beforeSha.slice(0, 7)}-${event.afterSha.slice(0, 7)}`;

      const group: CollapsedIterationGroup = {
        id: groupId,
        beforeSha: event.beforeSha,
        afterSha: event.afterSha,
        iterations: [],
        visibility: 'collapsed',
      };

      // Find discarded commits for this force-push
      // This is a simplified heuristic - real implementation would trace commit graph
      const groupDiscarded = discardedCommits.filter((c) => {
        // Check if commit was created before force-push and is not in live commits
        const isBeforeForcePush = c.createdAt <= event.timestamp;
        const isNotLive = !liveCommits.some((lc) => lc.sha === c.sha);
        return isBeforeForcePush && isNotLive && !commitToGroup.has(c.sha);
      });

      if (groupDiscarded.length === 0 && event.beforeSha) {
        group.unavailableReason = 'commits_unavailable';
      }

      for (const commit of groupDiscarded) {
        commitToGroup.set(commit.sha, groupId);
      }

      collapsedGroups.push(group);
    }

    // Combine and sort all commits by date
    const allCommits = [...liveCommits, ...discardedCommits].sort(
      (a, b) => a.createdAt.getTime() - b.createdAt.getTime()
    );

    // Remove duplicates (same SHA)
    const seenShas = new Set<string>();
    const uniqueCommits = allCommits.filter((c) => {
      if (seenShas.has(c.sha)) return false;
      seenShas.add(c.sha);
      return true;
    });

    // Assign sequential revision numbers
    let revision = 1;
    for (const commit of uniqueCommits) {
      const groupId = commitToGroup.get(commit.sha);
      const lineage = groupId ? 'discarded' : 'current';

      const iteration: StatelessIteration = {
        revision: revision++,
        commitSha: commit.sha,
        baseSha: commit.parentSha || prBaseSha,
        author: commit.author,
        message: commit.message,
        createdAt: commit.createdAt,
        lineage,
        collapsedGroupId: groupId,
      };

      iterations.push(iteration);

      // Add to collapsed group if discarded
      if (groupId) {
        const group = collapsedGroups.find((g) => g.id === groupId);
        if (group) {
          group.iterations.push(iteration);
        }
      }
    }

    span.addEvent('iterations.built', {
      [Attr.ITERATION_COUNT]: iterations.length,
      'collapsed_group.count': collapsedGroups.length,
    });
    span.setStatus('ok');

    return { iterations, collapsedGroups };
  } catch (error) {
    span.setStatus('error', error instanceof Error ? error.message : 'Unknown error');
    throw error;
  } finally {
    span.end();
  }
}
```

```typescript
    const span = tracer.startSpan('storage.getIterations', {
      [Attr.STORAGE_KEY]: prKey,
    });
    try {
      const index = this.db.transaction('iterations').store.index('by-prKey');
      const result = await index.getAll(prKey);
      span.addEvent('iterations.retrieved', { count: result.length });
      span.setStatus('ok');
      return result;
    } catch (error) {
      span.setStatus('error', error instanceof Error ? error.message : 'Unknown');
      throw error;
    } finally {
      span.end();
    }
  }

  async addIterations(records: IterationRecord[]): Promise<void> {
    const span = tracer.startSpan('storage.addIterations', {
      'record.count': records.length,
    });
    try {
      const tx = this.db.transaction('iterations', 'readwrite');
      let added = 0;

      for (const record of records) {
        // Check if exists (immutable - don't overwrite)
        const existing = await tx.store.get(record.key);
        if (!existing) {
          await tx.store.add(record);
          added++;
        }
      }

      await tx.done;
      span.addEvent('iterations.added', { added, skipped: records.length - added });
      span.setStatus('ok');
    } catch (error) {
      span.setStatus('error', error instanceof Error ? error.message : 'Unknown');
      throw error;
    } finally {
      span.end();
    }
  }

  async isUnavailable(prKey: string, revision: number): Promise<boolean> {
    const key = `${prKey}/${revision}`;
    const span = tracer.startSpan('storage.isUnavailable', {
      [Attr.STORAGE_KEY]: key,
    });
    try {
      const result = await this.db.get('unavailable', key);
      span.setStatus('ok');
      return result !== undefined;
    } catch (error) {
      span.setStatus('error', error instanceof Error ? error.message : 'Unknown');
      throw error;
    } finally {
      span.end();
    }
  }

  async markUnavailable(record: UnavailableRecord): Promise<void> {
    const span = tracer.startSpan('storage.markUnavailable', {
      [Attr.STORAGE_KEY]: record.key,
      'reason': record.reason,
    });
    try {
      await this.db.put('unavailable', record);
      span.setStatus('ok');
    } catch (error) {
      span.setStatus('error', error instanceof Error ? error.message : 'Unknown');
      throw error;
    } finally {
      span.end();
    }
  }

  close(): void {
    this.db.close();
  }
}

export async function openStatelessStorage(): Promise<StatelessStorage | null> {
  const span = tracer.startSpan('storage.open');

  try {
    const db = await openDB<StatelessDB>(DB_NAME, DB_VERSION, {
      upgrade(db) {
        // Last seen store
        if (!db.objectStoreNames.contains('lastSeen')) {
          db.createObjectStore('lastSeen', { keyPath: 'prKey' });
        }

        // Iterations store with index
        if (!db.objectStoreNames.contains('iterations')) {
          const iterStore = db.createObjectStore('iterations', { keyPath: 'key' });
          iterStore.createIndex('by-prKey', 'prKey');
        }

        // Unavailable store
        if (!db.objectStoreNames.contains('unavailable')) {
          db.createObjectStore('unavailable', { keyPath: 'key' });
        }
      },
    });

    span.setStatus('ok');
    span.end();
    return new IDBStatelessStorage(db);
  } catch (error) {
    // Graceful fallback for private browsing or IndexedDB disabled
    span.setStatus('error', 'IndexedDB unavailable');
    span.end();
    console.warn('IndexedDB unavailable, stateless persistence disabled:', error);
    return null;
  }
```

```typescript
// src/features/diff/scheduler/priority-queue.ts

export type Comparator<T> = (a: T, b: T) => number;

export class PriorityQueue<T> {
  private heap: T[] = [];

  constructor(private comparator: Comparator<T>) {}

  push(item: T): void {
    this.heap.push(item);
    this.bubbleUp(this.heap.length - 1);
  }

  pop(): T | undefined {
    if (this.heap.length === 0) return undefined;

    const result = this.heap[0];
    const last = this.heap.pop();

    if (this.heap.length > 0 && last !== undefined) {
      this.heap[0] = last;
      this.bubbleDown(0);
    }

    return result;
  }

  peek(): T | undefined {
    return this.heap[0];
  }

  size(): number {
    return this.heap.length;
  }

  clear(): void {
    this.heap = [];
  }

  remove(predicate: (item: T) => boolean): T | undefined {
    const index = this.heap.findIndex(predicate);
    if (index === -1) return undefined;

    const removed = this.heap[index];
    const last = this.heap.pop();

    if (index < this.heap.length && last !== undefined) {
      this.heap[index] = last;
      this.bubbleDown(index);
      this.bubbleUp(index);
    }

    return removed;
  }

  update(predicate: (item: T) => boolean, updater: (item: T) => T): boolean {
    const index = this.heap.findIndex(predicate);
    if (index === -1) return false;

    this.heap[index] = updater(this.heap[index]);
    this.bubbleDown(index);
    this.bubbleUp(index);
    return true;
  }

  private bubbleUp(index: number): void {
    while (index > 0) {
      const parentIndex = Math.floor((index - 1) / 2);
      if (this.comparator(this.heap[index], this.heap[parentIndex]) >= 0) break;

      [this.heap[index], this.heap[parentIndex]] = [this.heap[parentIndex], this.heap[index]];
      index = parentIndex;
    }
  }

  private bubbleDown(index: number): void {
    const length = this.heap.length;

    while (true) {
      const leftChild = 2 * index + 1;
      const rightChild = 2 * index + 2;
      let smallest = index;

      if (leftChild < length && this.comparator(this.heap[leftChild], this.heap[smallest]) < 0) {
        smallest = leftChild;
      }
      if (rightChild < length && this.comparator(this.heap[rightChild], this.heap[smallest]) < 0) {
        smallest = rightChild;
      }

      if (smallest === index) break;

      [this.heap[index], this.heap[smallest]] = [this.heap[smallest], this.heap[index]];
      index = smallest;
    }
  }
}
```

```typescript
// src/features/diff/workers/diff-compute.worker.ts
import * as Comlink from 'comlink';
import type { DiffComputeAPI, WorkerConfig } from './diff-compute.api';
import type { DiffTask, DiffResult, SpanTrackerResult } from '../scheduler/types';
import { computeLineDiff, computeAlignment } from './diff-engine';

// For tracing in worker context
function log(level: 'debug' | 'info' | 'warn' | 'error', name: string, attrs: Record<string, unknown>) {
  console[level](JSON.stringify({ timestamp: Date.now(), name, ...attrs }));
}

class DiffComputeWorkerImpl implements DiffComputeAPI {
  private config: WorkerConfig | null = null;
  private activeControllers = new Map<string, AbortController>();

  async init(config: WorkerConfig): Promise<void> {
    this.config = config;
    log('info', 'worker.init', { owner: config.owner, repo: config.repo });
  }

  async computeDiff(task: DiffTask): Promise<DiffResult> {
    const controller = new AbortController();
    this.activeControllers.set(task.taskId, controller);

    log('debug', 'worker.computeDiff.start', { taskId: task.taskId, filePath: task.filePath });

    try {
      const [leftContent, rightContent] = await Promise.all([
        this.fetchFileContent(task.filePath, task.leftSha, controller.signal),
        this.fetchFileContent(task.filePath, task.rightSha, controller.signal),
      ]);

      if (leftContent.unavailable || rightContent.unavailable) {
        return {
          taskId: task.taskId,
          status: 'unavailable',
          unavailableReason: leftContent.unavailable || rightContent.unavailable,
        };
      }

      const diffLines = computeLineDiff(leftContent.content!, rightContent.content!);
      const alignedLines = computeAlignment(diffLines);

      log('debug', 'worker.computeDiff.complete', {
        taskId: task.taskId,
        lineCount: diffLines.length
      });

      return {
        taskId: task.taskId,
        status: 'completed',
        diffLines,
        alignedLines,
      };
    } catch (error) {
      if (error instanceof DOMException && error.name === 'AbortError') {
        log('debug', 'worker.computeDiff.cancelled', { taskId: task.taskId });
        return { taskId: task.taskId, status: 'cancelled' };
      }

      log('error', 'worker.computeDiff.error', {
        taskId: task.taskId,
        error: error instanceof Error ? error.message : 'Unknown'
      });
      return {
        taskId: task.taskId,
        status: 'error',
        error: error instanceof Error ? error.message : 'Unknown error',
      };
    } finally {
      this.activeControllers.delete(task.taskId);
    }
  }

  async computeSpanTracker(task: DiffTask): Promise<SpanTrackerResult> {
    const controller = new AbortController();
    this.activeControllers.set(task.taskId, controller);

    log('debug', 'worker.computeSpanTracker.start', { taskId: task.taskId });

    try {
      const [leftContent, rightContent] = await Promise.all([
        this.fetchFileContent(task.filePath, task.leftSha, controller.signal),
        this.fetchFileContent(task.filePath, task.rightSha, controller.signal),
      ]);

      if (leftContent.unavailable || rightContent.unavailable) {
        return {
          taskId: task.taskId,
          status: 'error',
          error: `File unavailable: ${leftContent.unavailable || rightContent.unavailable}`,
        };
      }

      const diffLines = computeLineDiff(leftContent.content!, rightContent.content!);
      const mappings = this.buildLineMappings(diffLines);

      log('debug', 'worker.computeSpanTracker.complete', {
        taskId: task.taskId,
        mappingCount: mappings.length
      });

      return {
        taskId: task.taskId,
        status: 'completed',
        mappings,
      };
    } catch (error) {
      if (error instanceof DOMException && error.name === 'AbortError') {
        return { taskId: task.taskId, status: 'cancelled' };
      }

      return {
        taskId: task.taskId,
        status: 'error',
        error: error instanceof Error ? error.message : 'Unknown error',
      };
    } finally {
      this.activeControllers.delete(task.taskId);
    }
  }

  cancel(taskId: string): void {
    const controller = this.activeControllers.get(taskId);
    if (controller) {
      log('debug', 'worker.cancel', { taskId });
      controller.abort();
    }
  }

  private async fetchFileContent(
    path: string,
    sha: string,
    signal: AbortSignal
  ): Promise<{ content?: string; unavailable?: '404' | '410' }> {
    if (!this.config) throw new Error('Worker not initialized');

    const url = `https://api.github.com/repos/${this.config.owner}/${this.config.repo}/contents/${path}?ref=${sha}`;

    const headers: HeadersInit = {
      Accept: 'application/vnd.github+json',
    };
    if (this.config.token) {
      headers.Authorization = `Bearer ${this.config.token}`;
    }

    log('debug', 'worker.fetchFileContent', { path, sha });

    const response = await fetch(url, { headers, signal });

    if (!response.ok) {
      if (response.status === 404) return { unavailable: '404' };
      if (response.status === 410) return { unavailable: '410' };
      throw new Error(`GitHub API error: ${response.status}`);
    }

    const data = await response.json();

    if (data.encoding === 'base64') {
      return { content: atob(data.content.replace(/\n/g, '')) };
    }

    return { content: data.content };
  }

  private buildLineMappings(diffLines: unknown[]): Array<{
    leftLine: number;
    rightLine: number | null;
    type: 'unchanged' | 'modified' | 'added' | 'deleted';
  }> {
    // Simplified mapping - real implementation would use diff-engine output
    const mappings: Array<{
      leftLine: number;
      rightLine: number | null;
      type: 'unchanged' | 'modified' | 'added' | 'deleted';
    }> = [];

    let leftLine = 1;
    let rightLine = 1;

    for (const line of diffLines as Array<{ type: string }>) {
      if (line.type === 'equal') {
        mappings.push({ leftLine, rightLine, type: 'unchanged' });
        leftLine++;
        rightLine++;
      } else if (line.type === 'insert') {
        mappings.push({ leftLine: leftLine - 1, rightLine, type: 'added' });
        rightLine++;
      } else if (line.type === 'delete') {
        mappings.push({ leftLine, rightLine: null, type: 'deleted' });
        leftLine++;
      }
    }

    return mappings;
  }
}

// Factory function for testing
export function createWorkerAPI(): DiffComputeAPI {
  return new DiffComputeWorkerImpl();
}

// Expose for Comlink when running as actual worker
if (typeof self !== 'undefined' && 'postMessage' in self) {
  Comlink.expose(new DiffComputeWorkerImpl());
}
```

```typescript
// src/features/diff/scheduler/diff-scheduler.ts
import { tracer, SemanticAttributes as Attr } from '@/lib/tracing';
import { PriorityQueue } from './priority-queue';
import type {
  DiffTask,
  DiffResult,
  DiffPriority,
  ScheduleOptions,
  ScheduledTask,
  DiffScheduler,
} from './types';
import type { DiffComputeAPI } from '../workers/diff-compute.api';

const taskComparator = (a: ScheduledTask, b: ScheduledTask): number => {
  if (a.priority !== b.priority) return a.priority - b.priority;
  if (a.commentCount !== b.commentCount) return b.commentCount - a.commentCount;
  if (a.uiOrder !== b.uiOrder) return a.uiOrder - b.uiOrder;
  return a.enqueuedAt - b.enqueuedAt;
};

export function createDiffScheduler(worker: DiffComputeAPI): DiffScheduler {
  const queue = new PriorityQueue<ScheduledTask>(taskComparator);
  const cache = new Map<string, DiffResult>();
  const listeners = new Set<(result: DiffResult) => void>();

  let isProcessing = false;
  let currentTask: ScheduledTask | null = null;

  const processNext = async () => {
    if (isProcessing) return;

    const scheduled = queue.pop();
    if (!scheduled) return;

    isProcessing = true;
    currentTask = scheduled;

    const span = tracer.startSpan('scheduler.process', {
      [Attr.TASK_ID]: scheduled.task.taskId,
      [Attr.TASK_PRIORITY]: scheduled.priority,
      [Attr.DIFF_FILE_PATH]: scheduled.task.filePath,
    });

    try {
      const result = await worker.computeDiff(scheduled.task);

      cache.set(result.taskId, result);
      span.addEvent('task.completed', { status: result.status });

      for (const listener of listeners) {
        listener(result);
      }

      span.setStatus('ok');
    } catch (error) {
      span.setStatus('error', error instanceof Error ? error.message : 'Unknown');
    } finally {
      span.end();
      isProcessing = false;
      currentTask = null;

      // Process next in queue
      processNext();
    }
  };

  return {
    schedule(task: DiffTask, options: ScheduleOptions): string {
      const span = tracer.startSpan('scheduler.schedule', {
        [Attr.TASK_ID]: task.taskId,
        [Attr.TASK_PRIORITY]: options.priority,
      });

      const scheduled: ScheduledTask = {
        task,
        priority: options.priority,
        commentCount: options.commentCount ?? 0,
        uiOrder: options.uiOrder ?? 0,
        enqueuedAt: Date.now(),
      };

      queue.push(scheduled);
      span.addEvent('task.queued', { [Attr.TASK_QUEUE_DEPTH]: queue.size() });
      span.end();

      // Start processing if not already
      processNext();

      return task.taskId;
    },

    prioritize(taskId: string): void {
      const span = tracer.startSpan('scheduler.prioritize', {
        [Attr.TASK_ID]: taskId,
      });

      // Check if it's the current task
      if (currentTask?.task.taskId === taskId) {
        span.addEvent('task.already_processing');
        span.end();
        return;
      }

      // Update priority in queue
      const updated = queue.update(
        (t) => t.task.taskId === taskId,
        (t) => ({ ...t, priority: 0 }) // Highest = 0
      );

      if (updated) {
        span.addEvent('task.priority_bumped');

        // If there's a lower priority task processing, cancel it
        if (currentTask && currentTask.priority > 0) {
          span.addEvent('current_task.cancelling', {
            cancelledTaskId: currentTask.task.taskId
          });
          worker.cancel(currentTask.task.taskId);

          // Re-queue the cancelled task
          queue.push(currentTask);
        }
      }

      span.end();
    },

    clear(): void {
      const span = tracer.startSpan('scheduler.clear');

      queue.clear();
      cache.clear();

      if (currentTask) {
        worker.cancel(currentTask.task.taskId);
      }

      span.addEvent('scheduler.cleared');
      span.end();
    },

    getResult(taskId: string): DiffResult | undefined {
      return cache.get(taskId);
    },

    onComplete(callback: (result: DiffResult) => void): () => void {
      listeners.add(callback);
      return () => listeners.delete(callback);
    },
  };
}
```

```typescript
// src/features/diff/hooks/useStatelessDiff.ts
import { useState, useEffect, useMemo } from 'react';
import { useScheduler } from '../scheduler/context';
import { DiffPriority, type DiffResult } from '../scheduler';

interface UseStatelessDiffResult {
  status: 'loading' | 'completed' | 'error' | 'unavailable';
  diffLines?: unknown[];
  alignedLines?: unknown[];
  error?: string;
  unavailableReason?: '404' | '410';
}

export function useStatelessDiff(
  filePath: string,
  leftSha: string,
  rightSha: string,
  compareMode: '2dot' | '3dot' = '2dot'
): UseStatelessDiffResult {
  const scheduler = useScheduler();
  const [result, setResult] = useState<DiffResult | null>(null);

  const taskId = useMemo(
    () => `diff:${filePath}:${leftSha}:${rightSha}`,
    [filePath, leftSha, rightSha]
  );

  useEffect(() => {
    // Check cache first
    const cached = scheduler.getResult(taskId);
    if (cached) {
      setResult(cached);
      return;
    }

    // Schedule task
    scheduler.schedule(
      {
        taskId,
        type: 'compute_diff',
        filePath,
        leftSha,
        rightSha,
        compareMode,
      },
      { priority: DiffPriority.Highest }
    );

    // Subscribe to completion
    const unsubscribe = scheduler.onComplete((completedResult) => {
      if (completedResult.taskId === taskId) {
        setResult(completedResult);
      }
    });

    return unsubscribe;
  }, [scheduler, taskId, filePath, leftSha, rightSha, compareMode]);

  if (!result) {
    return { status: 'loading' };
  }

  if (result.status === 'unavailable') {
    return {
      status: 'unavailable',
      unavailableReason: result.unavailableReason,
    };
  }

  if (result.status === 'error') {
    return {
      status: 'error',
      error: result.error,
    };
  }

  return {
    status: 'completed',
    diffLines: result.diffLines,
    alignedLines: result.alignedLines,
  };
}
```
